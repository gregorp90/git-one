git init -- inside folder, initialize Git
git status -- see uncommitted files
git add <file-name> -- stage file
git add . -- stage all changed files
git rm --cached <file-name> -- unstage
git commit -m "<message>" -- commit changes
  (save point, like in a computer game, with
   a descriptive message what we changed)
git log -- see commit history
git log --oneline -- see commit history condensed
esc + :wq + enter
git checkout <commit-ID> -- look back in time
git checkout master -- go to current state
